[C#] event(イベント)

Study / C#    作成日付 : 2019/07/16 00:59:34   修正日付 : 2020/01/17 07:32:38

こんにちは。明月です。


今日はevent(イベント)について勉強します。


前の投稿で「delegate」はメソッドの動作を集まって呼び出せるものだと勉強しました。その中で「delegateフィールド」を通って「delegate」を格納あるいは呼び出すのができます。

でも「delegateフィールド」は「public」で公開すると内部だけではなくクラスの外部でも呼び出すのができます。

using System;
using System.Collections.Generic;
using System.Linq;
using System.Text;
using System.Threading.Tasks;

namespace eventEx
{
  class DelegateEx
  {
    public delegate void delegate1(String data);
    public delegate1 delegateField;
    public DelegateEx()
    {
      //delegate登録 - DelegateEx.delegateMethod
      delegateField += delegateMethod;
      //delegateを呼出す
      delegateField(" call DelegateEx");
    }
    public void delegateMethod(String data)
    {
      Console.WriteLine("class Inline " + data);
    }
  }
  class Program
  {
    public Program()
    {
      DelegateEx test = new DelegateEx();
      //delegate登録 - Program.delegateMethod
      test.delegateField += delegateMethod;
      //delegateを呼出す
      test.delegateField(" call Program");
    }
    public void delegateMethod(String data)
    {
      Console.WriteLine("class Outline " + data);
    }
    static void Main(string[] args)
    {
      new Program();
      Console.WriteLine("Press Any Key...");
      Console.ReadLine();
    }
  }
}

「delegate」はメソッドの代理子なので上のことに実装するのは悪いではないですが、そのことよりクラスの内部処理を外部に呼び出すのは怪しいじゃないかと思えられますね。

すなわち、「delegate」フィールドはクラスの内部処理が内部と外部のメソッドに呼び出すことができる型になります。


event(インベント)

「event」の場合はクラスの外部で呼び出すのを防止しています。使用方法は既存の「delegate」フィールドの前に「event」キーワードを付けることで使えます。

class
{
  アクセス修飾子 event デリゲート イベント名;
}


using System;
using System.Collections.Generic;
using System.Linq;
using System.Text;
using System.Threading.Tasks;

namespace eventEx
{
  class EventEx
  {
    public delegate void delegate1(String data);
    public event delegate1 eventField;
    public EventEx()
    {
      //event登録 - DelegateEx.delegateMethod
      eventField += delegateMethod;
      //delegateを呼出す
      eventField(" call DelegateEx");
    }
    public void delegateMethod(String data)
    {
      Console.WriteLine("class Inline " + data);
    }
  }
  class Program
  {
    public Program()
    {
      EventEx test = new EventEx();
      //event登録 - Program.delegateMethod
      test.eventField += delegateMethod;
      //eventを呼出す
      //外部で呼出すのはエラーになります。
      //test.eventField(" call Program");
    }
    public void delegateMethod(String data)
    {
      Console.WriteLine("class Outline " + data);
    }
    static void Main(string[] args)
    {
      new Program();
      Console.WriteLine("Press Any Key...");
      Console.ReadLine();
    }
  }
}

「delegateField」と「event」の差を纏めると下記とおりです。

  • ・delegateフィールドは使用演算子は(=,+=,-=)ですが、イベントは(+=,-=)だけです。
  • ・delegateフィールドはクラス内部、外部すべて呼び出すのができますが、eventはクラス内部だけ呼び出せます。
using System;
using System.Collections.Generic;
using System.Linq;
using System.Text;
using System.Threading.Tasks;

namespace eventEx
{
  class EventEx
  {
    public delegate void delegate1(String data);
    public event delegate1 eventField;
    public EventEx()
    {
      //event登録 - DelegateEx.delegateMethod
      eventField += delegateMethod;
      //delegateを呼出す
      eventField(" call DelegateEx");
    }
    public void delegateMethod(String data)
    {
      Console.WriteLine("class Inline " + data);
    }
  }
  class Program
  {
    public Program()
    {
      EventEx test = new EventEx();
      //event登録 - Program.delegateMethod
      test.eventField += delegateMethod;
      //eventを呼出す
      //外部で呼出すのはエラーになります。
      //test.eventField(" call Program");
    }
    public void delegateMethod(String data)
    {
      Console.WriteLine("class Outline " + data);
    }
    static void Main(string[] args)
    {
      new Program();
      Console.WriteLine("Press Any Key...");
      Console.ReadLine();
    }
  }
}
