  저번 포스팅까지 블로그 댓글 기능의 메인 부분은 전부 만들어졌습니다. 클라이언트(브라우저)에서 유저가 구글 아이디로 로그인하고 댓글을 작성하면 서버로 json 형태로 ajax 통신을 한다. 그리고 서버에서는 그 데이터를 받으면 데이터베이스에 저장한다. 클라이언트(브라우저)에서 유저가 해당 페이지의 댓글 리스트를 요청한다. 서버에서는 json형태로 작성하여 클라이언트로 댓글 정보를 준다. 이런 식으로 프로그램이 작성되었습니다. 아직 조그마한 버그가 눈에 띕니다만 그 부분은 포스팅 없이 수정했습니다.
앞으로 조금 더 사용하면서 불편한 점이 있으면 기능을 더욱 추가할 예정입니다만 먼저 추가 기능으로는 admin기능을 추가하려고 합니다.

  admin기능은 해당 댓글을 본 블로그 서비스(www.nowonbun.com)에서 관리하는 것이 아니고 서버 단에서 admin 페이지를 만들어서 관리할 예정입니다. 물론 딱히 관리페이지 없이 데이터베이스에서 직접 sql로 관리해도 상관은 없지만 불편하니 간단하게 로그인 기능 select, modify, 삭제 기능을 만들겠습니다.

        $clz->setParent ( $parent );
        $clz->setDeleted ( $isdeleted );

        array_push ( $rslt, $clz );
      }

        if($rslt[$i]->getParent() == null){
          array_push ( $list, $this->createJsonArray($rslt[$i]) );

      if ($stmt->execute ()) {
        parent::close();
      } else {
        parent::close();
        return array ("result" => "NG");
      }
      $stmt = null;
      $stmt = parent::getStmt ( "UPDATE COMMENT_T SET COMMENT=?, LASTUPDATED=now() WHERE IDX=?" );
      $stmt->bind_param ( "si", base64_encode ($comment), $idx);
      if ($stmt->execute ()) {
        parent::close();
        return array ("result" => "OK");
      } else {
        parent::close();
        return array ("result" => "NG");
      }
    } else if ($this->type == "DELETE" && strtoupper ( $_SERVER ['REQUEST_METHOD'] ) == 'POST' && $_SESSION["LOGIN"]){
      $idx = (int)trim($_REQUEST["idx"]);
      parent::validateIdx($idx);
      $stmt = null;
      $stmt = parent::getStmt ( "UPDATE COMMENT_T SET ISDELETED=1, LASTUPDATED=now() WHERE IDX=?" );
      $stmt->bind_param ( "i", $idx);
      if ($stmt->execute ()) {
        parent::close();
        return array ("result" => "OK");
      } else {
        parent::close();
        return array ("result" => "NG");
      }
    } else if ($this->type == "ACTIVE" && strtoupper ( $_SERVER ['REQUEST_METHOD'] ) == 'POST' && $_SESSION["LOGIN"]){
      $idx = (int)trim($_REQUEST["idx"]);
      parent::validateIdx($idx);
      $stmt = null;
      $stmt = parent::getStmt ( "UPDATE COMMENT_T SET ISDELETED=0, LASTUPDATED=now() WHERE IDX=?" );
      $stmt->bind_param ( "i", $idx);

        parent::close();
        return array ("result" => "OK");
      } else {
        parent::close();
        return array ("result" => "NG");
      }
    } else if(strtoupper ( $_SERVER ['REQUEST_METHOD'] ) == 'POST'){
      if(($this->pid == $this->sui && $this->pwd == $this->sup) || $_SESSION["LOGIN"]){
        $_SESSION["LOGIN"] = true;
        $this->login = true;
      } else {
        $this->error = "아이디, 패스워드를 확인해 주세요.";
      }
    } else {
      $_SESSION["LOGIN"] = null;
    }
  }
  private function getChild($list, $idx, $datalist){
    for($i=count($list)-1;$i>=0;$i--){
      if($idx == $list[$i]->getParent()){
        array_push ( $datalist, $this->createJsonArray($list[$i]) );
      }
    }
    return $datalist;
  }
  private function createJsonArray($node){
    return array (
      "idx" => $node->getIdx(),
      "url" => $node->getUrl(),
      "id" => $node->getId(),
      "ip" => $node->getIp(),
      "email" => $node->getEmail(),
      "comment" => $node->getComment(),
      "createdate" => $node->getCreatedate(),
      "lastupdated" => $node->getLastupdated(),
      "deleted" => $node->IsDeleted(),
      "parent" => $node->getParent()
    );
  }
  public function getId(){
    return $this->pid;
  }
  public function getPwd(){
    return $this->pwd;
  }
  public function getError(){
    return $this->error;
  }
  public function isLogin(){
    return $this->login;
  }
  public function getAjax() {
    return $this->ajax;
  }
}

    function commentAjax(data,cb){
      $.ajax({
        url: location.origin + location.pathname+"?"+data,
        type: "POST",
        dataType: "json",
        success: function (data, textStatus, jqXHR) {
          cb.call(this, data);
        },
        error: function (jqXHR, textStatus, errorThrown) {
          console.log(jqXHR);
          toastr.success("시스템 에러입니다.");
        },
        complete: function (jqXHR, textStatus) { }
      });
    }

  전체적인 설명을 하면 저는 페이지 소스(로그인, 리스트화면)와 ajax 소스(검색 데이터, 추가, 삭제)를 한 페이지에 담았습니다. 여러 페이지로 만들 수도 있겠지만, 관리페이지는 저만 사용한 것이고 딱히 여러페이지를 만들면 오히려 따로 소스를 관리해야 하지 않을까 하는 염려때문에 한페이지로 만들었습니다.
페이지 소스는 헤더를 「content-type: text/html;」로 설정하고 html 영역을 표시하는 방식으로 만들었고 ajax는 이전 상위 클래스 「AbstractDao」를 상속받아서 ajax형태를 그대로 유지하고 결과가 나오면 html영역은 die 함수를 써서 표시되지 않게 하였습니다.

  검색의 경우는 url별로 검색하는 것이 아니고 전체 댓글이 보일 수 있도록 수정하였습니다. 소스는 기존 등록, 검색, 삭제, 수정기능을 조금씩 수정해서 구현하였습니다.
  이로써 관리자 페이지도 완성했습니다. 특히 관리자 영역은 설명 없이 전반적인 설명과 소스를 나열하는 것도 포스팅을 작성했습니다.
이유는 댓글 기능의 포스팅 내용이 너무 길어지지 않나 생각도 되고, 연말연시까지 포스팅하면서 꼭 하고 싶은 작업이 있기 때문에 서둘러서 마무리했습니다. 아직 버그도 많고 또 추가하고 싶은 기능도 많이 있습니다만 그 부분에 대해서는 당장 작업하지는 않고 일단 사용하면서 불편한 점이나 추가하고 싶은 게 어느정도 쌓이면 한 번에 모아서 작업할 예정입니다.
