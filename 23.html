[C#] インタフェース( interface )

Study / C#    作成日付 : 2019/07/13 01:06:04   修正日付 : 2020/01/17 07:31:59

こんにちは。明月です。


今日はインタフェースについて勉強します。


今まで継承すると思えばクラスあるいは抽象クラスから引継いてもらってクラスを生成することでした。でもその形なら２つ以上の継承をできずに統一されたデータタイプに結ぶことができないですね。

各クラスは機能別で分けて処理するプロジェクトの案件がある場合、継承クラスをAType,BType,CType,DTypeを作成して実装します。

using System;
using System.Collections.Generic;
using System.Linq;
using System.Text;
using System.Threading.Tasks;

namespace InterfaceEx
{
  //抽象クラス Atype
  public abstract class Atype
  {
    public abstract void AProcess();
    public abstract void BProcess();
  }
  //抽象クラス Btype
  public abstract class Btype
  {
    public abstract void AProcess();
    public abstract void CProcess();
  }
  //抽象クラス Ctype
  public abstract class Ctype
  {
    public abstract void DProcess();
    public abstract void BProcess();
  }
  //抽象クラス Dtype
  public abstract class Dtype
  {
    public abstract void DProcess();
    public abstract void BProcess();
  }

  //一つしか継承できない。
  public class ATypeProcess : Atype
  {
    public override void AProcess()
    {
      Console.WriteLine("Atype AProcess");
    }
    public override void BProcess()
    {
      Console.WriteLine("Atype BProcess");
    }
  }
  //一つしか継承できない。
  public class CTypeProcess : Ctype
  {
    public override void DProcess()
    {
      Console.WriteLine("Ctype DProcess");
    }
    public override void BProcess()
    {
      Console.WriteLine("Ctype BProcess");
    }
  }
  class Program
  {
    static void Main(string[] args)
    {
      //抽象をしましたが、纏めができない。
      //割り当てはATypeProcessでしましたが親クラス(Atype)のタイプで定義ができる。
      Atype atype = new ATypeProcess();
      atype.AProcess();
      atype.BProcess();
      //抽象をしましたが、纏めができない。
      //割り当てはCTypeProcessでしましたが親クラス(Ctype)のタイプで定義ができる。
      Ctype ctype = new CTypeProcess();
      ctype.DProcess();
      ctype.BProcess();

      Console.WriteLine("Press Any Key...");
      Console.ReadLine();
    }
  }
}

でも上の場合は一つずつ区分ができますが２つ以上に機能を結ぶのはできないですね。そのため、多重継承ができるものが必要でそれをインタフェース(interface)と言います。


インタフェース(interface)

インタフェース(Interface)は抽象体なので実装しているロジックがあることではなく、抽象メソッドだけあります。

using System;
using System.Collections.Generic;
using System.Linq;
using System.Text;
using System.Threading.Tasks;

namespace InterfaceEx
{
  //インタフェース
  public interface Atype
  {
    void AProcess();
  }
  public interface Btype
  {
    void BProcess();
  }
  public interface Ctype
  {
    void CProcess();
  }
  public interface Dtype
  {
    void DProcess();
  }

  //インタフェースは多重継承ができる。(Atype,Btype)
  public class ATypeProcess : Atype, Btype
  {
    public void AProcess()
    {
      Console.WriteLine("Atype AProcess");
    }
    public void BProcess()
    {
      Console.WriteLine("Atype BProcess");
    }
  }
  //インタフェースは多重継承ができる。(Btype,Dtype)
  public class CTypeProcess : Btype, Dtype
  {
    public void DProcess()
    {
      Console.WriteLine("Ctype DProcess");
    }
    public void BProcess()
    {
      Console.WriteLine("Ctype BProcess");
    }
  }
  class Program
  {
    static void Main(string[] args)
    {
      ATypeProcess atype = new ATypeProcess();
      atype.AProcess();
      atype.BProcess();
      CTypeProcess ctype = new CTypeProcess();
      ctype.BProcess();
      ctype.DProcess();

      Console.WriteLine();
      //インタフェースがBtypeの共通インタフェースになるから結ぶのができます。
      List<Btype> list = new List<Btype>();
      list.Add(atype);
      list.Add(ctype);
      foreach (Btype item in list)
      {
        item.BProcess();
      }

      Console.WriteLine("Press Any Key...");
      Console.ReadLine();
    }
  }
}

上の例を見るとインタフェースを使われなくても、プログラム作成することができそうですね。ですから、インタフェースを使わなくても問題がなさそうだし、性能にも影響がなさそうですね。

そのいうとおりでインタフェースを知らなくてもプログラムを作成できます。でもプロジェクトをデザイン、すなわちコーディング設計をする時にはインタフェースがなければ実装が大変になります。


例えば、プロジェクトを構成する時にDBが２つがあると考えてみましょう。２つのデータベースで構成されているテーブル(データ構造)も全て違うことで、ユーザの場合は２つのデータを選ぶことではなく一つのデータベースから取得するみたいに構成して提供しましょうという課題があると考えましょう。

using System;
using System.Collections.Generic;
using System.Linq;
using System.Text;
using System.Threading.Tasks;

namespace InterfaceEx
{
  /// <summary>
  /// テーブル区分する列挙型
  /// </summary>
  public enum TABLE
  {
    ATABLE,
    BTABLE,
    CTABLE,
    DTABLE
  }
  /// <summary>
  /// データベースインタフェース
  /// </summary>
  public interface Database
  {
    DataTable getTable(TABLE tableName);
  }
  /// <summary>
  /// データテーブルインタフェース
  /// </summary>
  public interface DataTable
  {
    String getData();
  }
  /// <summary>
  /// ATABLEのデータテーブル
  /// </summary>
  public class ATABLE : DataTable
  {
    public String getData()
    {
      return "ATABLE";
    }
  }
  /// <summary>
  /// BTABLEのデータテーブル
  /// </summary>
  public class BTABLE : DataTable
  {
    public String getData()
    {
      return "BTABLE";
    }
  }
  /// <summary>
  /// CTABLEのデータテーブル
  /// </summary>
  public class CTABLE : DataTable
  {
    public String getData()
    {
      return "CTABLE";
    }
  }
  /// <summary>
  /// DTABLEのデータテーブル
  /// </summary>
  public class DTABLE : DataTable
  {
    public String getData()
    {
      return "DTABLE";
    }
  }
  /// <summary>
  /// Aデータベース
  /// </summary>
  public class ADatabase : Database 
  {
    public DataTable getTable(TABLE tableName)
    {
      //AデータベースはATableとBTableを持っています。
      if(tableName == TABLE.ATABLE)
      {
        return new ATABLE();
      }else if(tableName == TABLE.BTABLE)
      {
        return new BTABLE();
      }
      return null;
    }
  }
  public class BDatabase : Database
  {
    public DataTable getTable(TABLE tableName)
    {
      //AデータベースはCTableとDTableを持っています。
      if (tableName == TABLE.CTABLE)
      {
        return new CTABLE();
      }
      else if (tableName == TABLE.DTABLE)
      {
        return new DTABLE();
      }
      return null;
    }
  }

  class Program
  {
    static void Main(string[] args)
    {
      //データベース宣言してリストに格納する。
      List<Database> DBList = new List<Database>();
      DBList.Add(new ADatabase());
      DBList.Add(new BDatabase());

      //データテーブル取得
      //ここからはテーブルがADatabaseにあるかBDatabaseにあるかは関係ない。
      //ただあるところで取得します。
      //すなわち、データベースは二つだけどユーザはデータベースには関係なく、データだけ扱っている。(OOPの特徴)
      DataTable table = getTable(DBList, TABLE.CTABLE);
      String ret = table.getData();
      Console.WriteLine(ret);

      Console.WriteLine("Press Any Key...");
      Console.ReadLine();
    }
    //データテーブル取得メソッド
    static DataTable getTable(List<Database> DBList, TABLE tableName)
    {
      foreach(Database Item in DBList)
      {
        DataTable dt = Item.getTable(tableName);
        if(dt != null)
        {
          return dt;
        }
      }
      return null;
    }
  }
}

上の例を見ると全体構造は少し複雑になりましたが、ロジックを使うユーザ(Main関数)はデータベースを気にしずにデータだけ考えてロジックを実装ができます。

また、上の条件ならデータベースが２つではなくもっと追加になってもソース修正が要らずにインタフェースに合わせてDBListに追加することができます。


プロジェクトは一人で立ち上げることよりチームを組んで立ち上げる時が多いです。その時にモジュール実装するたびにモジュール間に機能影響があったり、エラーが発生するとプロジェクトを進めることが大変になると思います。

それでOOP開発はモジュールの結合分離することで勧めます。そのためにOOPを実装するためにはインタフェースが必要なことです。
