LINQ(リンク) - メソッド

前の投稿にはLINQを「SQLQuery」タイプで使いました。筆者もソースの可読性のため「SQLQuery」タイプをお勧めしますが、仕様によって検索条件が複雑になると「SQLQuery」だけですべての条件を実装できないと思います。

そのためクエリタイプのLINQではなく、メソッドタイプの「LINQ」があって実装できます。


WHEREメソッド

﻿using System;
using System.Collections.Generic;
using System.Linq;
using System.Text;
using System.Threading.Tasks;

namespace LINQEX2
{
  //テストデータ
  class LINQExampleData
  {
    //コンストラクタ
    public LINQExampleData(int number, String data)
    {
      this.Number = number;
      this.Data = data;
    }
    public int Number
    {
      get; set;
    }
    public String Data
    {
      get; set;
    }
  }
  class Program
  {
    static void Main(string[] args)
    {
      //リストにテストデータを格納する。
      List<LINQExampleData> list = new List<LINQExampleData>();
      list.Add(new LINQExampleData(1, "AAAA"));
      list.Add(new LINQExampleData(2, "BBBB"));
      list.Add(new LINQExampleData(3, "CCCC"));
      list.Add(new LINQExampleData(4, "DDDD"));
      list.Add(new LINQExampleData(5, "EEEE"));
      list.Add(new LINQExampleData(6, "FFFF"));
      list.Add(new LINQExampleData(7, "GGGG"));
      list.Add(new LINQExampleData(8, "HHHH"));
      list.Add(new LINQExampleData(9, "IIII"));

      //WHEREメソッド
      IEnumerable<LINQExampleData> ret =  list.Where(
        //メソッド形式で複雑な処理を作成できます。
        delegate (LINQExampleData item) {
          if (item.Number > 4)
          {
            return true;
          }
          else
          {
            return false;
          }
        });

      ret = ret.Where(
        //条件処理を二つを連結して作られる。
        delegate (LINQExampleData item) {
          if (item.Number < 8)
          {
            return true;
          }
          else
          {
            return false;
          }
        });

      //結果変換
      IEnumerator<LINQExampleData> iter = ret.GetEnumerator();
      //結果はNumberが4超過8未満
      while (iter.MoveNext())
      {
        Console.WriteLine(iter.Current.Data);
      }

      Console.WriteLine("Press Any Key...");
      Console.ReadLine();
    }
  }
}

上の例を見るとList(リスト)で「Where」メソッドを使っています。そのメソッド中でデリゲートの匿名メソッドを使って繰り返して返却値が「true」場合にデータを出すようになっています。

すなわち、上の例は初めの「Where」メソッドで「Number」の値が5以上のデータかつ２つ目の「Where」メソッドで「Number」が８の以下のデータを返却してデータを出力します。


Orderメソッド

using System;
using System.Collections.Generic;
using System.Linq;
using System.Text;
using System.Threading.Tasks;

namespace LINQEX2
{
  //テストデータ
  class LINQExampleData
  {
    //コンストラクタ
    public LINQExampleData(int number, String data)
    {
      this.Number = number;
      this.Data = data;
    }
    public int Number
    {
      get; set;
    }
    public String Data
    {
      get; set;
    }
  }
  class Program
  {
    static void Main(string[] args)
    {
      //リストにテストデータを格納する。
      List<LINQExampleData> list = new List<LINQExampleData>();
      list.Add(new LINQExampleData(1, "AAAA"));
      list.Add(new LINQExampleData(2, "BBBB"));
      list.Add(new LINQExampleData(3, "CCCC"));
      list.Add(new LINQExampleData(4, "DDDD"));
      list.Add(new LINQExampleData(5, "EEEE"));
      list.Add(new LINQExampleData(6, "FFFF"));
      list.Add(new LINQExampleData(7, "GGGG"));
      list.Add(new LINQExampleData(8, "HHHH"));
      list.Add(new LINQExampleData(9, "IIII"));
      //リストの順番処理 (Descの降順)
      IEnumerable<LINQExampleData> ret = list.OrderByDescending(
        delegate (LINQExampleData item)
        {
          return item.Number;
        });

      //結果変換
      IEnumerator<LINQExampleData> iter = ret.GetEnumerator();
      //結果
      while (iter.MoveNext())
      {
        Console.WriteLine(iter.Current.Data);
      }

      Console.WriteLine("Press Any Key...");
      Console.ReadLine();
    }
  }
}

上の例は「LINQ」の「Orderby」の降順のメソッドです。Where関数と同じみたいにメソッド中には匿名メソッドで処理して、返却値は順番対象の値を返却します。

すなわち、「LINQExampleData」の「Number」の降順になって「9,8,7,6」になります。出力はデータの値を出力します。


SELECTメソッド

using System;
using System.Collections.Generic;
using System.Linq;
using System.Text;
using System.Threading.Tasks;

namespace LINQEX2
{
  //テストデータ
  class LINQExampleData
  {
    //コンストラクタ

    {
      this.Number = number;
      this.Data = data;
    }
    public int Number
    {
      get; set;
    }
    public String Data
    {
      get; set;
    }
  }
  class Program
  {
    static void Main(string[] args)
    {
      //リストにテストデータを格納する。
      List<LINQExampleData> list = new List<LINQExampleData>();
      list.Add(new LINQExampleData(1, "AAAA"));
      list.Add(new LINQExampleData(2, "BBBB"));
      list.Add(new LINQExampleData(3, "CCCC"));
      list.Add(new LINQExampleData(4, "DDDD"));
      list.Add(new LINQExampleData(5, "EEEE"));
      list.Add(new LINQExampleData(6, "FFFF"));
      list.Add(new LINQExampleData(7, "GGGG"));
      list.Add(new LINQExampleData(8, "HHHH"));
      list.Add(new LINQExampleData(9, "IIII"));

      //リストの順番処理 (Descの降順)
      IEnumerable<LINQExampleData> ret = list.OrderByDescending(
        delegate (LINQExampleData item)
        {
          return item.Number;
        });
        
      //リストの返却値を「LINQExampleData」タイプから「String」タイプに変換する。
      IEnumerable<String>ret2 = ret.Select(
        delegate (LINQExampleData item)
        {
          return item.Number + " " + item.Data;
        });
      //結果変換
      IEnumerator<String> iter = ret2.GetEnumerator();
      //結果
      while (iter.MoveNext())
      {
        Console.WriteLine(iter.Current);
      }

      Console.WriteLine("Press Any Key...");
      Console.ReadLine();
    }
  }
}

上の例の場合は「Select」メソッドを利用して「LINQExampleData」タイプから「String」タイプに変換していますね。


Joinメソッド

using System;
using System.Collections.Generic;
using System.Linq;
using System.Text;
using System.Threading.Tasks;

namespace LINQEX2
{
  //テストデータ
  class LINQExampleData
  {
    //コンストラクタ
    public LINQExampleData(int number, String data)
    {
      this.Number = number;
      this.Data = data;
    }
    public int Number
    {
      get; set;
    }
    public String Data
    {
      get; set;
    }
  }
  class Program
  {
    static void Main(string[] args)
    {
      //リストにテストデータを格納する。
      List<LINQExampleData> list = new List<LINQExampleData>();
      list.Add(new LINQExampleData(1, "AAAA"));
      list.Add(new LINQExampleData(2, "BBBB"));
      list.Add(new LINQExampleData(3, "CCCC"));
      list.Add(new LINQExampleData(4, "DDDD"));
      list.Add(new LINQExampleData(5, "EEEE"));
      list.Add(new LINQExampleData(6, "FFFF"));
      list.Add(new LINQExampleData(7, "GGGG"));
      list.Add(new LINQExampleData(8, "HHHH"));
      list.Add(new LINQExampleData(9, "IIII"));

      List<LINQExampleData> list2 = new List<LINQExampleData>();
      list2.Add(new LINQExampleData(5, "JOIN data AAAA"));
      list2.Add(new LINQExampleData(6, "JOIN data BBBB"));

      //list2のリストから6だけ取得
      IEnumerable<LINQExampleData> list2Enumerable = list2.Where(
        delegate (LINQExampleData item)
        {
          return item.Number == 6;
        });
      //list1のリストを降順に変換する。
      IEnumerable<LINQExampleData> ret = list.OrderByDescending(
        delegate (LINQExampleData item)
        {
          return item.Number;
        });
      //list2のリストの「Number」が６のリストとlistデータと「Join」する
      ret = list.Join(list2Enumerable, 
        delegate (LINQExampleData listItem)
        {
          return listItem.Number;
        },
        delegate (LINQExampleData list2Item)
        {
          return list2Item.Number;
        },
        delegate (LINQExampleData listItem, LINQExampleData list2Item)
        {
          listItem.Data = listItem.Data + " " + list2Item.Data;
          return listItem;
        });
      //「Join」の結果で「Number」と「Data」をあわせた「String」データを返却
      IEnumerable<String> ret2 = ret.Select(
        delegate (LINQExampleData item)
        {
          return item.Number + " " + item.Data;
        });
      //結果変換
      IEnumerator<String> iter = ret2.GetEnumerator();
      //結果はNumberが4超過8未満
      while (iter.MoveNext())
      {
        Console.WriteLine(iter.Current);
      }

      Console.WriteLine("Press Any Key...");
      Console.ReadLine();
    }
  }
}

上の例は２つのリストを組み合わせて結果を出力することです。様々の処理が入りましたが、「Join」メソッドだけ見るとデリゲートが３つがあることが確認できます。

初めのパラメタは組み合わせるリスト、２つ目は対象レコードで比較するデータ、３つ目は組み合わせるリストの比較するデータ、最後は組み合わせたデータを返却するデリゲートになっています。
